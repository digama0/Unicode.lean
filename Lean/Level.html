def Lean.Level.isNeverZero :
Lean.Level → Bool

If result is true, then forall assignments A which assigns all parameters and metavariables occuring in l, l[A] != zero

Equations
  • Lean.Level.isNeverZero (Lean.Level.zero x) = false
  • Lean.Level.isNeverZero (Lean.Level.param x x_1) = false
  • Lean.Level.isNeverZero (Lean.Level.mvar x x_1) = false
  • Lean.Level.isNeverZero (Lean.Level.succ x x_1) = true
  • Lean.Level.isNeverZero (Lean.Level.max l₁ l₂ x) = (Lean.Level.isNeverZero l₁ || Lean.Level.isNeverZero l₂)
  • Lean.Level.isNeverZero (Lean.Level.imax l₁ l₂ x) = Lean.Level.isNeverZero l₂
